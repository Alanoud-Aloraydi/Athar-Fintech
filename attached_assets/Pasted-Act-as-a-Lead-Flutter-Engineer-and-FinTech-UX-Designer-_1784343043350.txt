### 1. Update the JS Bridge (`assets/oasis/oasis_viewer.html`)
Rewrite the HTML/JS bridge to handle BOTH "Growth" (Visibility of Palms 1 to 9) and "Health" (CSS Visual Filters for dryness). Use this EXACT script:

```html
<!DOCTYPE html>
<html>
<head>
  <meta charset="utf-8" />
  <style>
    html, body { margin: 0; height: 100%; background: transparent; overflow: hidden; }
    #canvas3d { width: 100%; height: 100%; display: block; transition: filter 2s ease-in-out; }
  </style>
</head>
<body>
  <canvas id="canvas3d"></canvas>
  <script type="module">
    import { Application } from '[https://unpkg.com/@splinetool/runtime@1.9.82/build/runtime.js](https://unpkg.com/@splinetool/runtime@1.9.82/build/runtime.js)';

    const SCENE_URL = '[https://prod.spline.design/oa7Ns7PvmCdHvQCP/scene.splinecode](https://prod.spline.design/oa7Ns7PvmCdHvQCP/scene.splinecode)';
    const TOTAL_PALMS = 9;
    const canvas = document.getElementById('canvas3d');
    const spline = new Application(canvas);

    function applyGrowth(progressPct) {
      const visibleCount = Math.max(1, Math.min(TOTAL_PALMS, 1 + Math.floor((progressPct * 100) / 12.5)));
      for (let i = 1; i <= TOTAL_PALMS; i++) {
        const name1 = 'Palm_0' + i;
        const name2 = 'Palm_' + i;
        let obj = spline.findObjectByName(name1) || spline.findObjectByName(name2);
        if (obj) { obj.visible = (i <= visibleCount); }
      }
    }

    function applyHealth(healthScore) {
      if (healthScore >= 80) { canvas.style.filter = 'none'; } 
      else if (healthScore >= 50) { canvas.style.filter = 'saturate(0.7)'; } 
      else { canvas.style.filter = 'saturate(0.3) sepia(0.35)'; }
    }

    spline.load(SCENE_URL).then(() => {
      applyGrowth(0.0); 
      window.parent.postMessage({ event: 'ready' }, '*');
    });

    window.addEventListener('message', (e) => {
      if (e.data && e.data.cmd === 'updateOasisState') {
        applyGrowth(e.data.progress);
        applyHealth(e.data.health);
      }
    });
  </script>
</body>
</html>
